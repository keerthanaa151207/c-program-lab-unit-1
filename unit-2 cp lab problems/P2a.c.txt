//Bubble sort
#include<stdio.h>
int main(){
    int a[]={5,3,2,1,6,4};
    int n=6,i,j,temp;
    printf("before sorting\n");
    for(i=0;i<n;i++)
    printf("%d\n",a[i]);
    for(i=0;i<n-1;i++)
    {
        for(j=0;j<n-i-1;j++)
        { if(a[j]>a[j+1])
            {temp=a[j];
             a[j]=a[j+1];
             a[j+1]=temp;}}}
        printf("after sorting\n");
        for(i=0;i<n;i++)
        printf("%d\n",a[i]);
        return 0;
}
