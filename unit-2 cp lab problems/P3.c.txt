// c program to insert an element into an sorted array
#include<stdio.h>
int main()
{
    int a[5]={10,20,30,40,50};
    int number,n=5,i;
    printf("enter a number to insert\n");
    scanf("%d",&number);
    for(i=n-1;i>=0 && a[i]>number;i--)
    {a[i+1]=a[i];}
    a[i+1]=number;
    n=n+1;
    printf("after insertion\n");
    for(i=0;i<n;i++)
    printf("%d\n",a[i]);
    return 0;
}
