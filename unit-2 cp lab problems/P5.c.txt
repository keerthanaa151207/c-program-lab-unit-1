//program for linear search
#include<stdio.h>
int main()
{
    int arr[5]={10,20,30,40,50};
    int n=30;
    int i,found=0;
    for(i=0;i<5;i++)
    {
        if(arr[i]==n)
        { found=1;
        break;}
    }
    if(found)
    printf("element found at index %d",i);
    else
    printf("elements not found");
    return 0;
}
